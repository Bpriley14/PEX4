function main() {
    
    int user_count = 10;
    const string WELCOME_MESSAGE = "Welcome to the program!";

    
    greet();

    
    if (user_count > 5) {
        print(WELCOME_MESSAGE);
    } else {
        print("Too few users.");
    }

    
    int counter = 5;
    while (counter > 0) {
        print(counter);
        counter = counter - 1;
    }
}

function void greet() {
    print("Hello, User!");
}
